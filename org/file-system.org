(evil-define-command +evil:move-this-file (new-path &optional force-p)
  "Move current buffer's file to NEW-PATH. Replaces %, # and other vim-esque
filename modifiers (see `+evil*ex-replace-special-filenames'). If FORCE-P,
overwrite the destination file if it exists, without confirmation."
  :repeat nil
  (interactive "<f><!>")
  (when (or (not new-path) (string-empty-p new-path))
    (user-error "No new path was specified"))
  (doom/move-this-file new-path force-p))

(defun config/rename-current-buffer-file-vc ()
  (interactive)
  (let* ((old-path (buffer-file-name))
         (new-path (read-file-name "Rename file to: "
                                   (file-name-directory old-path)

                                   (file-name-nondirectory old-path))))
    (if (vc-backend old-path)
        (vc-rename-file old-path new-path)

    (set-visited-file-name new-path t t)))

(defun doom-files--update-refs (&rest files)
  "Ensure FILES are updated in `recentf', `magit' and `save-place'."
  (let (toplevels)
    (dolist (file files)
      (when (featurep 'vc)
        (vc-file-clearprops file)
        (when-let (buffer (get-file-buffer file))
          (with-current-buffer buffer
            (vc-refresh-state))))
      (when (featurep 'magit)
        (when-let (default-directory (magit-toplevel (file-name-directory file)))
          (cl-pushnew default-directory toplevels)))
      (unless (file-readable-p file)
        (when (bound-and-true-p recentf-mode)
          (recentf-remove-if-non-kept file))
        (when (and (bound-and-true-p projectile-mode)
                   (doom-project-p)
                   (projectile-file-cached-p file (doom-project-root)))
          (projectile-purge-file-from-cache file))))
    (dolist (default-directory toplevels)
      (magit-refresh))
    (when (bound-and-true-p save-place-mode)
      (save-place-forget-unreadable-files))))

(defun doom/move-this-file (new-path &optional force-p)
  "Move current buffer's file to NEW-PATH.

If FORCE-P, overwrite the destination file if it exists, without confirmation."
  (interactive
   (list (read-file-name "Move file to: ")
         current-prefix-arg))
  (unless (and buffer-file-name (file-exists-p buffer-file-name))
    (user-error "Buffer is not visiting any file"))
  (let ((old-path (buffer-file-name (buffer-base-buffer)))
        (new-path (expand-file-name new-path)))
    (when (directory-name-p new-path)
      (setq new-path (concat new-path (file-name-nondirectory old-path))))
    (make-directory (file-name-directory new-path) 't)
    (rename-file old-path new-path (or force-p 1))
    (set-visited-file-name new-path t t)
    (doom-files--update-refs old-path new-path)
    (message "File moved to %S" (abbreviate-file-name new-path))
    )
  )

(defun config/current-buffer-file-delete-vc ()
  (interactive)
  (let ((file (buffer-file-name)))
    (if (vc-backend file)
        (progn (vc-delete-file file) (kill-buffer))
      (when (yes-or-no-p (format "%s is not under version control. Delete permanently? "
                                 (file-name-nondirectory file)))
        (delete-file file)
        (kill-buffer)))))

(defun config/current-buffer-file-delete-no-vc ()
  (interactive)
  (let ((file (buffer-file-name)))
    (when (yes-or-no-p (format "Permanently delete %s? " (file-name-nondirectory file)))
      (delete-file file)
      (kill-buffer))))

(defun config/chmod-current-buffer-file ()
  "Change permissions of the current buffer's file.
Accepts octal (e.g. 755) or symbolic chmod expressions (e.g. u+x, o-w, a=r).
Shows current permissions in the prompt; offers common modes as completion candidates."
  (interactive)
  (let* ((file (buffer-file-name))
         (_ (unless file (user-error "Buffer is not visiting a file")))
         (current-mode (file-modes file))
         (current-octal (format "%o" current-mode))
         (current-symbolic (file-modes-number-to-symbolic current-mode))
         (candidates '("644" "755" "600" "700" "664" "775" "440" "444" "400"
                       "u+x" "u-x" "a+x" "a-x" "o+w" "o-w" "g+w" "g-w"))
         (input (completing-read
                 (format "chmod [now: %s %s]: " current-octal current-symbolic)
                 candidates nil nil nil nil current-octal))
         (new-mode (if (string-match-p "^[0-7]+$" input)
                       (string-to-number input 8)
                     (file-modes-symbolic-to-number input current-mode))))
    (set-file-modes file new-mode)
    (message "Permissions of %s set to %s (%s)"
             (file-name-nondirectory file)
             (format "%o" new-mode)
             (file-modes-number-to-symbolic new-mode))))

;; Path yanking functions
(defun config/yank-filename ()
  "Copy the current buffer's filename (without directory) to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((name (file-name-nondirectory filename)))
        (kill-new name)
        (message "Copied filename: %s" name))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-absolute-path ()
  "Copy the current buffer's absolute file path to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (progn
        (kill-new filename)
        (message "Copied absolute path: %s" filename))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-home-relative-path ()
  "Copy the current buffer's path with ~ substituted for home to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((path (abbreviate-file-name filename)))
        (kill-new path)
        (message "Copied home-relative path: %s" path))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-relative-path ()
  "Copy the current buffer's path relative to `default-directory' to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((relative (file-relative-name filename default-directory)))
        (kill-new relative)
        (message "Copied relative path: %s" relative))
    (user-error "Buffer is not visiting a file")))

(defun config/--project-root ()
  "Return the current project root, or nil if not in a project."
  (or (when-let (proj (and (fboundp 'project-current) (project-current)))
        (project-root proj))
      (when (fboundp 'projectile-project-root)
        (ignore-errors (projectile-project-root)))))

(defun config/yank-project-relative-path ()
  "Copy the current buffer's path relative to the project root to the kill ring.
Errors if the buffer is not in a project."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((root (or (config/--project-root)
                      (user-error "Not in a project"))))
        (let ((relative (file-relative-name filename root)))
          (kill-new relative)
          (message "Copied project-relative path: %s" relative)))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-dwim-path ()
  "Copy the current buffer's project-relative path, falling back to `default-directory'."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let* ((root (or (config/--project-root) default-directory))
             (relative (file-relative-name filename root)))
        (kill-new relative)
        (message "Copied path: %s" relative))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-relative-directory ()
  "Copy the current buffer's relative directory path (no filename) to the kill ring.
Relative to project root if in a project, otherwise to `default-directory'."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let* ((root (or (and (fboundp 'projectile-project-root)
                            (projectile-project-root))
                       (and (fboundp 'doom-project-root)
                            (doom-project-root))
                       default-directory))
             (dir (file-name-directory filename))
             (relative (file-relative-name dir root)))
        (kill-new relative)
        (message "Copied relative directory: %s" relative))
    (user-error "Buffer is not visiting a file")))

(defun config/copy-current-buffer-file (new-path &optional stay)
  "Copy current buffer's file to NEW-PATH.
If STAY is non-nil, keep the current buffer open; otherwise switch to the copy."
  (let* ((old-path (buffer-file-name))
         (_ (unless old-path (user-error "Buffer is not visiting a file")))
         (new-path (expand-file-name new-path)))
    (when (directory-name-p new-path)
      (setq new-path (concat new-path (file-name-nondirectory old-path))))
    (copy-file old-path new-path 1)
    (if stay
        (message "Copied to %s" (abbreviate-file-name new-path))
      (find-file new-path))))

(defun config/copy-current-buffer-file-and-switch ()
  "Copy current buffer's file to a new path and switch to the copy."
  (interactive)
  (let* ((old-path (buffer-file-name))
         (_ (unless old-path (user-error "Buffer is not visiting a file")))
         (new-path (read-file-name "Copy file to: "
                                   (file-name-directory old-path)
                                   nil nil
                                   (file-name-nondirectory old-path))))
    (config/copy-current-buffer-file new-path nil)))

(defun config/copy-current-buffer-file-keep-original ()
  "Copy current buffer's file to a new path, staying on the original."
  (interactive)
  (let* ((old-path (buffer-file-name))
         (_ (unless old-path (user-error "Buffer is not visiting a file")))
         (new-path (read-file-name "Copy file to: "
                                   (file-name-directory old-path)
                                   nil nil
                                   (file-name-nondirectory old-path))))
    (config/copy-current-buffer-file new-path t)))

(defun config/open-current-buffer-file-externally ()

  (if-let ((file (buffer-file-name)))
      (shell-command (concat "open " (shell-quote-argument file)))
    (user-error "Buffer is not visiting a file")))

  "Reveal the current buffer's file in macOS Finder."
  (interactive)
  (if-let ((file (buffer-file-name)))
      (shell-command (concat "open -R " (shell-quote-argument file)))
    (user-error "Buffer is not visiting a file")))

(defun config/sudo-edit-current-buffer-file ()
  "Reopen the current buffer's file with sudo via TRAMP."
  (interactive)
  (if-let ((file (buffer-file-name)))
      (find-file (concat "/sudo::" (expand-file-name file)))
    (user-error "Buffer is not visiting a file")))
