(evil-define-command +evil:move-this-file (new-path &optional force-p)
  "Move current buffer's file to NEW-PATH. Replaces %, # and other vim-esque
filename modifiers (see `+evil*ex-replace-special-filenames'). If FORCE-P,
overwrite the destination file if it exists, without confirmation."
  :repeat nil
  (interactive "<f><!>")
  (when (or (not new-path) (string-empty-p new-path))
    (user-error "No new path was specified"))
  (doom/move-this-file new-path force-p))

(defun config/rename-current-buffer-file-vc ()
  (interactive)
  (let* ((old-path (buffer-file-name))
         (new-path (read-file-name "Rename file to: "
                                   (file-name-directory old-path)
                                   nil nil
                                   (file-name-nondirectory old-path))))
    (vc-rename-file old-path new-path)
    (set-visited-file-name new-path t t)))

(defun doom-files--update-refs (&rest files)
  "Ensure FILES are updated in `recentf', `magit' and `save-place'."
  (let (toplevels)
    (dolist (file files)
      (when (featurep 'vc)
        (vc-file-clearprops file)
        (when-let (buffer (get-file-buffer file))
          (with-current-buffer buffer
            (vc-refresh-state))))
      (when (featurep 'magit)
        (when-let (default-directory (magit-toplevel (file-name-directory file)))
          (cl-pushnew default-directory toplevels)))
      (unless (file-readable-p file)
        (when (bound-and-true-p recentf-mode)
          (recentf-remove-if-non-kept file))
        (when (and (bound-and-true-p projectile-mode)
                   (doom-project-p)
                   (projectile-file-cached-p file (doom-project-root)))
          (projectile-purge-file-from-cache file))))
    (dolist (default-directory toplevels)
      (magit-refresh))
    (when (bound-and-true-p save-place-mode)
      (save-place-forget-unreadable-files))))

(defun doom/move-this-file (new-path &optional force-p)
  "Move current buffer's file to NEW-PATH.

If FORCE-P, overwrite the destination file if it exists, without confirmation."
  (interactive
   (list (read-file-name "Move file to: ")
         current-prefix-arg))
  (unless (and buffer-file-name (file-exists-p buffer-file-name))
    (user-error "Buffer is not visiting any file"))
  (let ((old-path (buffer-file-name (buffer-base-buffer)))
        (new-path (expand-file-name new-path)))
    (when (directory-name-p new-path)
      (setq new-path (concat new-path (file-name-nondirectory old-path))))
    (make-directory (file-name-directory new-path) 't)
    (rename-file old-path new-path (or force-p 1))
    (set-visited-file-name new-path t t)
    (doom-files--update-refs old-path new-path)
    (message "File moved to %S" (abbreviate-file-name new-path))
    )
  )

(defun config/current-buffer-file-delete-vc ()
  (interactive)
  (let ((file (buffer-file-name)))
    (vc-delete-file file)
    (kill-buffer)))

(defun config/current-buffer-file-delete-no-vc ()
  (interactive)
  (let ((file (buffer-file-name)))
    (delete-file file)
    (kill-buffer)))

;; Path yanking functions
(defun config/yank-filename ()
  "Copy the current buffer's filename (without directory) to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((name (file-name-nondirectory filename)))
        (kill-new name)
        (message "Copied filename: %s" name))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-absolute-path ()
  "Copy the current buffer's absolute file path to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (progn
        (kill-new filename)
        (message "Copied absolute path: %s" filename))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-relative-path ()
  "Copy the current buffer's relative file path to the kill ring.
Relative to project root if in a project, otherwise to `default-directory'."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let* ((root (or (and (fboundp 'projectile-project-root)
                            (projectile-project-root))
                       (and (fboundp 'doom-project-root)
                            (doom-project-root))
                       default-directory))
             (relative (file-relative-name filename root)))
        (kill-new relative)
        (message "Copied relative path: %s" relative))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-relative-directory ()
  "Copy the current buffer's relative directory path (no filename) to the kill ring.
Relative to project root if in a project, otherwise to `default-directory'."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let* ((root (or (and (fboundp 'projectile-project-root)
                            (projectile-project-root))
                       (and (fboundp 'doom-project-root)
                            (doom-project-root))
                       default-directory))
             (dir (file-name-directory filename))
             (relative (file-relative-name dir root)))
        (kill-new relative)
        (message "Copied relative directory: %s" relative))
    (user-error "Buffer is not visiting a file")))

(defun config/yank-absolute-directory ()
  "Copy the current buffer's absolute directory path (no filename) to the kill ring."
  (interactive)
  (if-let ((filename (buffer-file-name)))
      (let ((dir (file-name-directory filename)))
        (kill-new dir)
        (message "Copied absolute directory: %s" dir))
    (user-error "Buffer is not visiting a file")))

(eval-after-load "evil"
  (progn
    (map!
     :leader
     :prefix ("f" . "files")
     :desc "File rename" "r" #'config/rename-current-buffer-file-vc
     :desc "File rename (no-vc)" "R" #'config/rename-current-buffer-file-no-vc
     :desc "File delete" "d" #'config/current-buffer-file-delete-vc
     :desc "File delete (no VC)" "D" #'config/current-buffer-file-delete-no-vc
     (:prefix ("p" . "path")
      :desc "Yank filename" "n" #'config/yank-filename
      :desc "Yank absolute path" "a" #'config/yank-absolute-path
      :desc "Yank relative path" "r" #'config/yank-relative-path
      :desc "Yank relative directory" "d" #'config/yank-relative-directory
      :desc "Yank absolute directory" "D" #'config/yank-absolute-directory)
     )
    ;; This keeps eval-after-load from calling the last thing evaluated (which
    ;; would be the function passed to `map!').
    nil
    )
  )
